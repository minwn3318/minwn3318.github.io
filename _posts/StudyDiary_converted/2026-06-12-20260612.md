```
#include <iostream>

void FormatSetting() {
	// c언어에서 사용하는 출력 매서드, iostream 헤더파일을 include 하여 출력한다
	printf("hello! \n");
	printf("\n");

	// 공백 상태에서는 오류 발생
	//printf();

	// 변수로도 출력할 수 있다
	int a = 10;
	double b = 0.5;
	char c = 'A';
	char d[] = "name";

	printf("%d개 \n", a);
	printf("%f확률 \n", b);
	printf("알파벳의 첫 글자, %c \n", c);
	printf("my %s is... \n", d);
	printf("\n");

	// 서식 지정자 연습
	// 정수
	int aa = -15;
	unsigned int bb = 12;

	printf("%d\n", aa);
	printf("%i\n", aa);
	printf("%u\n", bb);
	printf("%o\n", bb);
	printf("%x\n", bb);
	printf("%X\n", bb);
	printf("\n");

	// 실수
	float cc = 3.140000;
	double dd = 3.140000e+00;
	long double ee = 3.14;

	printf("%f\n", cc);
	printf("%f\n", dd);
	printf("%.3f\n", cc);
	printf("%.1f\n", dd);
	printf("%lf\n", dd);
	printf("%Lf\n", ee);
	printf("%e\n", dd);
	printf("%E\n", dd);
	printf("\n");

	//문자열
	char ff = 'a';
	char gg[] = "hello";
	char hh[] = "hello world!";
	char ii[] = "hello wo\0rld!";

	printf("%c\n", ff);
	printf("%s\n", gg);
	printf("%s\n", hh);
	printf("%s\n", ii);
	printf("\n");

	//주소
	printf("%p\n", &aa);
	printf("%p\n", &bb);
	printf("\n");

	//길이지정자
	short shor = -1;
	unsigned short unsho = 7;
	char cha = 2;
	unsigned char uncha = 5;
	long lon = -5000000;
	unsigned long unlong = 4000000;
	wchar_t wcha = 8;
	long long llong = -598000000;
	unsigned long long unll = 800000000;
	long double ld = -1.00052318500;

	printf("%hd\n", shor);
	printf("%hu\n", unsho);
	printf("%hhd\n", cha);
	printf("%hhu\n", uncha);
	printf("%ld\n", lon);
	printf("%lu\n", unlong);
	printf("%lc\n", wcha);
	printf("%lld\n", llong);
	printf("%llu\n", unll);
	printf("%Lf\n", ld);
	printf("\n");

	//특수출력문자
	printf("\n hello\n");
	printf("\t hello\n");
	printf("\\hello/\n");
	printf("\"hello\"\n");
	printf("\'hello\'\n");
	printf("%%hello%%");

}
```
